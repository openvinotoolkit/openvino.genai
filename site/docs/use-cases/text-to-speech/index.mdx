<ConvertModelSection>

    <OptimumCLI model='microsoft/speecht5_tts' outputDir='speecht5_tts' weightFormat='int4' trustRemoteCode params={{
        "--model-kwargs": JSON.stringify({ vocoder: "microsoft/speecht5_hifigan" })
    }} />
