    auto rerank_result = pipeline.rerank(query, documents);
    
    std::cout << "Reranked documents:\\n";
    for (const auto& [index, score] : rerank_result) {
        std::cout << "Document " << index << " (score: " << score << "): " << documents[index] << '\\n';
    }
}
