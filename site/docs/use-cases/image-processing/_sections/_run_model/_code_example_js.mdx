{`import fs from "node:fs/promises";
import { extname, join } from "node:path";
import { addon as ov } from "openvino-node";
import { VLMPipeline } from "openvino-genai-node";
import jpegJs from "jpeg-js";
import { PNG } from "pngjs";

async function readImage(imagePath) {
    const buffer = await fs.readFile(imagePath);
    const ext = extname(imagePath).toLowerCase();
    let width, height, rgbaData;
    if (ext === ".jpg" || ext === ".jpeg") {
        ({ width, height, data: rgbaData } = jpegJs.decode(buffer, { useTArray: true }));
    } else if (ext === ".png") {
        ({ width, height, data: rgbaData } = PNG.sync.read(buffer));
    } else {
        throw new Error(\`Unsupported image format: \${ext}\`);
    }
    const pixelCount = width * height;
    const rgb = new Uint8Array(pixelCount * 3);
    for (let i = 0; i < pixelCount; i++) {
        rgb[i * 3]     = rgbaData[i * 4];
        rgb[i * 3 + 1] = rgbaData[i * 4 + 1];
        rgb[i * 3 + 2] = rgbaData[i * 4 + 2];
    }
    return new ov.Tensor(ov.element.u8, [height, width, 3], rgb);
}

async function readImages(imagePath) {
    const stats = await fs.stat(imagePath);
    if (stats.isDirectory()) {
        const files = await fs.readdir(imagePath);
        return Promise.all(files.sort().map((file) => readImage(join(imagePath, file))));
    }
    return [await readImage(imagePath)];
}
