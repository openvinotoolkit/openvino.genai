import { readImageWithFfmpeg } from "./ffmpeg_utils.js";

async function readImages(imagePath) {
    const stats = await stat(imagePath);
    if (stats.isDirectory()) {
        const files = await readdir(imagePath);
        return Promise.all(files.sort().map((file) => readImageWithFfmpeg(path.join(imagePath, file))));
    }
    return [await readImageWithFfmpeg(imagePath)];
}
